CREATE table medications(
	id SERIAL PRIMARY KEY
    );
    
create table medication_class(
	id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    medication_id int references medications(id)
	);
    
create table drugs(
	id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    dose varchar(100),
    strength varchar(100),
    medication_class_id int references medication_class(id)
    );
